/**
 * @fileoverview Middleware store pentru procesarea și rutarea mesajelor WebSocket
 * 
 * Acest middleware:
 * 1. Procesează mesajele primite conform structurii definite în RESPONSE_DOCUMENTATION.md
 * 2. Normalizează și validează datele primite.
 * 3. Rutează acțiunile către store-ul `useChatStore` pentru a controla UI-ul (mesaje, view curent, overlay-uri).
 * 4. Oferă un singur punct de acces pentru procesarea răspunsurilor de la WebSocket.
 */
 
import { create } from 'zustand';
import { useChatStore } from './chatStore';
import { 
    RESPONSE_TYPES, 
    CHAT_INTENTS,
    INCOMING_MESSAGE_TYPES
} from '../actions/types';
 
// Import handlers
import { handleReservationsUpdate } from '../actions/handlers/reservationHandler';
import { handleHistory } from '../actions/handlers/historyHandler';
import { handleNotification } from '../actions/handlers/automationHandler';
 
// TODO: Remove this temporary import if reservation store is separate
import { useCalendarStore } from './calendarStore';
 
// Funcție helper pentru a extrage numele componentei dintr-o acțiune/intent de tip "show_*"
// e.g., "show_calendar" -> "calendar"
const getComponentNameFromShowAction = (showAction) => {
    if (showAction && typeof showAction === 'string' && showAction.startsWith('show_')) {
        const componentName = showAction.substring(5);
        return componentName || null; // Returnează null dacă e gol după 'show_'
    } 
    return null;
};
 
// Funcție helper pentru a mapa un intent la tipul de overlay corespunzător
// Acesta este tipul folosit de componenta ChatOverlay
const mapIntentToOverlayType = (intent) => {
    switch (intent) {
        case CHAT_INTENTS.RESERVATION:
        case CHAT_INTENTS.MODIFY_RESERVATION:
            return 'reservation'; // Corespunde OVERLAY_TYPES.RESERVATION din ChatOverlay
        case CHAT_INTENTS.CREATE_ROOM:
        case CHAT_INTENTS.MODIFY_ROOM:
            return 'room'; // Corespunde OVERLAY_TYPES.ROOM_MANAGEMENT din ChatOverlay
        case CHAT_INTENTS.SELL_PRODUCT:
            return 'productSales'; // Corespunde OVERLAY_TYPES.PRODUCT_SALES din ChatOverlay
        // Adaugă alte mapări dacă sunt necesare
        default:
            console.warn(`[MIDDLEWARE] mapIntentToOverlayType: Intent '${intent}' cannot be mapped to a known overlay type.`);
            return null;
    }
};
 
const useMiddlewareStore = create((set, get) => ({
  /**
   * Funcția principală pentru procesarea mesajelor primite de la WebSocketWorker.
   * Interpretează structura conform RESPONSE_DOCUMENTATION.md și apelează acțiunile din useChatStore.
   * De asemenea, gestionează actualizările de date primite de la server (rezervări, istoric, notificări).
   */
  processMessage: (event) => {
    if (!event || !event.data) {
      console.error("❌ [MIDDLEWARE] Invalid event received");
      return;
    }
 
    console.group("📩 [MIDDLEWARE] Processing message");
    console.log("Raw event data:", event.data);
    
    // Extragem datele comune. Numele pot varia ușor.
    const messageType = event.data.type; // Folosim un nume generic aici
    const { intent, action, message, payload: dataPayload, data, notification } = event.data;
 
    // Validăm tipul mesajului principal
    // Trebuie să verificăm atât RESPONSE_TYPES cât și INCOMING_MESSAGE_TYPES
    const isValidResponseType = Object.values(RESPONSE_TYPES).includes(messageType);
    const isValidIncomingType = Object.values(INCOMING_MESSAGE_TYPES).includes(messageType);
 
    if (!isValidResponseType && !isValidIncomingType) {
        console.error(`❌ [MIDDLEWARE] Invalid or missing message type received: ${messageType}`);
        console.groupEnd();
        useChatStore.getState().addMessage({
            type: 'error',
            text: `System error: Received an unknown message type from server (${messageType || 'missing'}).`,
        });
        return;
    }
 
    console.log(`Processing: type=${messageType}, intent=${intent || 'N/A'}, action=${action || 'N/A'}`);
 
    // Preluăm acțiunile necesare din useChatStore
    const { addMessage, setLatestIntent, setDisplayComponent, showOverlay } = useChatStore.getState();
    // TODO: Importați și acțiunile din alte store-uri (reservations, history, notifications) când sunt implementate
    // --- TEMPORARY --- Get actions for reservation handler
    const { setReservations } = useCalendarStore.getState();
 
    // --- Procesare specifică bazată pe Tipul Mesajului --- 
 
    // Folosim messageType (care este event.data.type) pentru switch
    switch (messageType) {
        // Cazuri existente pentru RESPONSE_TYPES (UI Commands from Chat)
        case RESPONSE_TYPES.SECONDARY:
            console.log(`[MIDDLEWARE] Handling SECONDARY: action=${action}`);
            const componentFromSecondary = getComponentNameFromShowAction(action);
            if (componentFromSecondary) {
                setDisplayComponent(componentFromSecondary);
                console.log(`[MIDDLEWARE] Set display component to: '${componentFromSecondary}'`);
            } else {
                console.warn(`[MIDDLEWARE] SECONDARY response type received with invalid or missing 'action': ${action}. Cannot set display component.`);
                addMessage({ type: 'error', text: `System error: Received unclear instruction to switch view (${action || 'missing action'}).` });
            }
            break;
 
        case RESPONSE_TYPES.OVERLAY:
            console.log(`[MIDDLEWARE] Handling OVERLAY: intent=${intent}, action=${action}`);
            
            // a. Setăm componenta de context (view-ul principal)
            const componentFromOverlayAction = getComponentNameFromShowAction(action);
            if (componentFromOverlayAction) {
                setDisplayComponent(componentFromOverlayAction);
                console.log(`[MIDDLEWARE] Set context component for overlay to: '${componentFromOverlayAction}'`);
            } else {
                console.warn(`[MIDDLEWARE] OVERLAY response received with invalid or missing context 'action': ${action}. Cannot set context component.`);
            }
 
            // b. Determinăm tipul și datele pentru overlay
            const overlayType = mapIntentToOverlayType(intent);
            let overlayData = dataPayload; // Folosim direct payload-ul imbricat conform documentației
 
            if (overlayType) {
                if (overlayData) {
                    // Adaptăm structura datelor dacă e necesar pentru componenta specifică
                    // TODO: Review and confirm data structure expected by each overlay component
                    if (overlayType === 'room') {
                        if (typeof overlayData === 'object' && overlayData !== null && (overlayData.id || overlayData.number)) {
                           overlayData = { room: overlayData }; 
                           console.log(`[MIDDLEWARE] Adapting data for 'room' overlay:`, overlayData);
                        } else {
                            console.error(`[MIDDLEWARE] Data for 'room' overlay seems incorrect. Expected room object, got:`, dataPayload);
                            overlayData = null; 
                            addMessage({ type: 'error', text: `System error: Received invalid data for room details.` });
                        }
                    } else if (overlayType === 'productSales') {
                         if (dataPayload && dataPayload.productName) {
                            overlayData = {
                                products: [{ name: dataPayload.productName, quantity: dataPayload.quantity || 1 }],
                                reservation: event.data.reservation // TODO: Clarify source
                            };
                            console.log(`[MIDDLEWARE] Adapting data for 'productSales' overlay:`, overlayData);
                         } else {
                            console.error(`[MIDDLEWARE] Data for 'productSales' overlay seems incorrect. Expected productName, got:`, dataPayload);
                            overlayData = null;
                            addMessage({ type: 'error', text: `System error: Received invalid data for product sale.` });
                         }
                    } else if (overlayType === 'reservation') {
                        if (typeof overlayData === 'object' && overlayData !== null && (overlayData.id || overlayData.fullName || overlayData.startDate)) {
                            console.log(`[MIDDLEWARE] Using data directly for 'reservation' overlay:`, overlayData);
                        } else {
                            console.error(`[MIDDLEWARE] Data for 'reservation' overlay seems incorrect. Expected reservation object, got:`, dataPayload);
                            overlayData = null;
                            addMessage({ type: 'error', text: `System error: Received invalid data for reservation details.` });
                        }
                    }
 
                    // c. Afișăm overlay-ul dacă avem tip și date valide
                    if (overlayData) {
                       console.log(`[MIDDLEWARE] Calling showOverlay: type='${overlayType}', data=`, overlayData);
                       showOverlay(overlayType, overlayData);
                    } else {
                         console.log(`[MIDDLEWARE] Not showing overlay due to invalid/adapted data.`);
                    }
 
                } else {
                    console.warn(`[MIDDLEWARE] OVERLAY type received for intent '${intent}' but missing nested 'payload' data.`);
                    addMessage({ type: 'error', text: `System error: Received request to open overlay ('${intent}') but crucial data is missing.` });
                }
            } else {
                console.warn(`[MIDDLEWARE] OVERLAY type received but intent '${intent}' could not be mapped to a known overlay type.`);
                addMessage({ type: 'error', text: `System error: Received unclear instruction to open an overlay ('${intent || 'missing intent'}).` });
            }
            break;
 
        case RESPONSE_TYPES.CHAT:
            // Mesajul este adăugat indiferent dacă are text sau nu (ex: mesaje de sistem fără text vizibil)
             if (message) {
                addMessage({
                    type: 'assistant', 
                    text: message,
                });
                console.log(`[MIDDLEWARE] Added chat message: "${message}"`);
            } else {
                 console.log(`[MIDDLEWARE] Received CHAT type message without text content.`);
            }
           
             // Setăm ultimul intent (chiar și pentru mesaje simple, poate fi util pentru context)
             if (intent && intent !== CHAT_INTENTS.DEFAULT) {
                setLatestIntent(intent);
                console.log(`[MIDDLEWARE] Set latest intent to: ${intent} from CHAT response`);
            }
            break;
 
        // Cazuri noi pentru INCOMING_MESSAGE_TYPES (Server Data Pushes)
        case INCOMING_MESSAGE_TYPES.RESERVATIONS:
            console.log(`[MIDDLEWARE] Handling RESERVATIONS: action=${action}`, data);
            // TODO: Call function from reservationStore to update reservations
            // Example: useReservationStore.getState().updateReservations(data?.reservations, action);
            if (!data || !data.reservations) {
                 console.warn("[MIDDLEWARE] RESERVATIONS message received without 'data.reservations'.");
            } else {
                // Call the specific handler
                // TODO: Confirm the correct way to pass store actions. Using calendarStore temporarily.
                handleReservationsUpdate(data, { setReservations });
            }
            break;
 
        case INCOMING_MESSAGE_TYPES.HISTORY:
            console.log("[MIDDLEWARE] Handling HISTORY:", data);
            if (!data || !data.items) {
                 console.warn("[MIDDLEWARE] HISTORY message received without 'data.items'.");
            } else {
                // Call the specific handler
                handleHistory(data); // Pass the whole data object as payload
            }
            break;
 
        case INCOMING_MESSAGE_TYPES.NOTIFICATION:
            console.log("[MIDDLEWARE] Handling NOTIFICATION:", notification);
            if (!notification) {
                console.warn("[MIDDLEWARE] NOTIFICATION message received without 'notification' data.");
            } else {
                // Call the specific handler
                handleNotification(event.data); 
            }
            break;
 
        default:
            // Acest caz nu ar trebui atins datorită validării de la început
            console.error(`[MIDDLEWARE] Reached default case in switch, which should not happen. messageType: ${messageType}`);
            break;
    }
 
    console.groupEnd();
  },
  
  // Funcția sendMessage rămâne neschimbată, trimite mesaje formatate către worker
  sendMessage: (message, worker) => {
      if (!worker) {
        console.error("❌ [MIDDLEWARE] WebSocket worker not available");
        return false;
      }
      
      try {
        let content;
        if (typeof message === 'string') {
          content = message;
        } else if (message && typeof message === 'object' && message.content) {
          content = message.content;
        } else {
          console.warn("⚠️ [MIDDLEWARE] Mesajul trimis nu este string sau obiect cu 'content'. Încercăm conversia la JSON.", message);
          try {
            content = JSON.stringify(message);
          } catch (e) {
            console.error("❌ [MIDDLEWARE] Nu s-a putut converti mesajul în string JSON.", message, e);
            return false;
          }
        }
  
        if (typeof content !== 'string' || content.trim() === '') {
            console.error("❌ [MIDDLEWARE] Conținutul mesajului este invalid sau gol. Mesajul nu va fi trimis.", content);
            return false;
        }
        
        const formattedMessage = {
          type: "CHAT_MESSAGE", // Trimitem doar acest tip conform specificațiilor anterioare
          content: content
        };
        
        console.log("📤 [MIDDLEWARE] Sending message via worker:", formattedMessage);
        
        worker.postMessage({
          type: "send_message", 
          payload: formattedMessage
        });
        
        return true;
      } catch (error) {
        console.error("❌ [MIDDLEWARE] Error sending message:", error);
        // Adăugăm mesaj de eroare în UI
        useChatStore.getState().addMessage({
          type: 'error',
          text: 'Failed to send message to server. Please check connection or try again.'
        });
        return false;
      }
  },
  
}));
 
export default useMiddlewareStore;
